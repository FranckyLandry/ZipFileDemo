namespace LaptopProduction
{
    public class Laptop
    {
        public string Name { get; set; }
        public string Msg { get; set; }
        public IPart Cpu { get; set; }
        public IPart Memory { get; set; }
        public IPart Graphics { get; set; }
        public IPart Storage { get; set; }
        public ILaptopFactory LaptopFactory { get; }

        public Laptop(ILaptopFactory laptopFactory)
        {
            LaptopFactory = laptopFactory;
        }

        public void BuildLaptop()
        {
            Cpu = LaptopFactory.CreateCpu();
            Msg += "CPU: " + Cpu.Name + "\n\n";
            Memory = LaptopFactory.CreateMemory();
            Msg += "Memory: " + Memory.Name + "\n\n";
            Graphics = LaptopFactory.CreateGraphics();
            Msg += "Video Card: " + Graphics.Name + "\n\n";
            Storage = LaptopFactory.CreateStorage();
            Msg += "Hard Drive: " + Storage.Name + "\n\n";
        }

        public void Assemble()
        {
            Msg += "Assembling your " + Name + "\n\n";
        }

        public void Packaging()
        {
            Msg += "Packaging your " + Name + "\n\n";
        }

        public override string ToString()
        {
            Msg += "Product completed!";
            return Msg;
        }
    }
}

namespace LaptopProduction
{
    public enum LaptopType
    {
        Economical, HighEnd, Gaming
    }

    public class LaptopStore
    {
        public Laptop Laptop { get; private set; }

        public LaptopStore()
        {
            Laptop = null;
        }

        public void OrderLaptop(LaptopType type)
        {
            switch (type)
            {
                case LaptopType.Economical:
                    Laptop = new EconomicalLaptop(new EconomicalFactory());
                    break;
                case LaptopType.HighEnd:
                    Laptop = new HighEndLaptop(new HighEndFactory());
                    break;
                case LaptopType.Gaming:
                    Laptop = new GamingLaptop(new GamingFactory());
                    break;
            }
            Laptop.BuildLaptop();
            Laptop.Assemble();
            Laptop.Packaging();
        }
    }
}

namespace LaptopProduction
{
    public abstract class Producer
    {
        public string Name { get; set; }
        public string ProgressMsg { get; set; }
        public ICpu Cpu { get; set; }
        public IMemory Memory { get; set; }
        public IGraphics Graphics { get; set; }
        public IDiscreteGraphics DescreteGraphics { get; set; }
        public IStorage Storage { get; set; }
        public ISsd Ssd { get; set; }
        public IFactory LaptopFactory { get; }

        protected Producer(IFactory laptopFactory)
        {
            LaptopFactory = laptopFactory;
        }

        public void PrepareCommonParts()
        {
            Cpu = LaptopFactory.CreateCpu();
            ProgressMsg += "CPU: " + Cpu.Name + "\n\n";
            Memory = LaptopFactory.CreateMemory();
            ProgressMsg += "Memory: " + Memory.Name + "\n\n";
            Graphics = LaptopFactory.CreateHdGraphics();
            ProgressMsg += "Video Card: " + Graphics.Name + "\n\n";
        }

        public abstract void PrepareSpecificParts();

        public void Assemble()
        {
            ProgressMsg += "Assembling your " + Name + "\n\n";
        }

        public void Packaging()
        {
            ProgressMsg += "Packaging your " + Name + "\n\n";
        }

        public override string ToString()
        {
            ProgressMsg += "Product completed!";
            return ProgressMsg;
        }
    }
}
