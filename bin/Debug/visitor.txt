        private void btnCalculate_Click(object sender, EventArgs e)
        {

            if (radioButton1.Checked)
            {
                visitor  = new TaxVisitor();
                obj.Accept(visitor);
                richTextBox1.Text = " ";

                if (type == "alcohol")
                {
                    richTextBox1.Text += visitor.visit(new Alcohol(15, "Vodka"));
                    richTextBox1.Text += visitor.visit(new Alcohol(15, "Baileys"));

                }
               else if(type == "chocolate")
                {
                    richTextBox1.Text += visitor.visit(new Chocolate(11, "White Truffle Chocolate"));
                    richTextBox1.Text += visitor.visit(new Chocolate(5, "Ferrero"));
                }
                else if(type == "parfume")
                {
                    richTextBox1.Text += visitor.visit(new Parfume(50, "Dolce and Gabbana"));
                    richTextBox1.Text += visitor.visit(new Parfume(30, "Justin Bieber"));
                }
                

            }
            else if (radioButton2.Checked)
            {
                visitor = new TaxVacationVisitor();
                obj.Accept(visitor);
                richTextBox1.Text = " ";

                if (type == "alcohol")
                {
                    richTextBox1.Text += visitor.visit(new Alcohol(15, "Vodka"));
                    richTextBox1.Text += visitor.visit(new Alcohol(15, "Baileys"));

                }
                else if (type == "chocolate")
                {
                    richTextBox1.Text += visitor.visit(new Chocolate(11, "White Truffle Chocolate"));
                    richTextBox1.Text += visitor.visit(new Chocolate(5, "Ferrero"));
                }
                else if (type == "parfume")
                {
                    richTextBox1.Text += visitor.visit(new Parfume(50, "Dolce and Gabbana"));
                    richTextBox1.Text += visitor.visit(new Parfume(30, "Justin Bieber"));
                }

            }
            else
            {
                MessageBox.Show("Select one type of tax please.");
            }

            panel1.Visible = false;
            pBAlchohol.BorderStyle = BorderStyle.None;
            pBChocolate.BorderStyle = BorderStyle.None;
            pBParfume.BorderStyle = BorderStyle.None;

        }


        private void pBAlchohol_Click(object sender, EventArgs e)
        {

            type = "alcohol";

        }
