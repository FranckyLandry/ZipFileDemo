    public partial class Form1 : Form
    {
        AbstractFactory MinionFactory, SmurfFactory;
        ICharacter MinionCharacter, SmurfCharacter;
        IHouse MinionHouse, SmurfHouse;
        IFood MinionFood, SmurfFood;

        public Form1()
        {
            InitializeComponent();

            MinionFactory = new MinionFactory();
            MinionCharacter = MinionFactory.CreateCharacter();
            MinionFood = MinionFactory.CreateFood();
            MinionHouse = MinionFactory.CreateHouse();

            SmurfFactory = new SmurfFactory();
            SmurfCharacter = SmurfFactory.CreateCharacter();
            SmurfFood = SmurfFactory.CreateFood();
            SmurfHouse = SmurfFactory.CreateHouse();
        }   

        private void btnGoShoppingM_Click(object sender, EventArgs e)
        {
            try
            {
                lbDescriptionM.Text = MinionFood.GoShopping(Convert.ToInt32(tbAmountM.Text));
            }
            catch (Exception)
            {
                MessageBox.Show("You have to write the amount of money!");
            }
        }

        private void btnCookM_Click(object sender, EventArgs e)
        {
            lbDescriptionM.Text = MinionFood.Cook();
        }

        private void btnBuildM_Click(object sender, EventArgs e)
        {
            pbM.Image = Image.FromFile("../Resources/minionhouse.jpg");
            lbDescriptionM.Text = MinionHouse.Build();
        }


        private void btnDecorateM_Click(object sender, EventArgs e)
        {
            lbDescriptionM.Text = MinionHouse.Decorate();
        }

        private void btnCleanM_Click(object sender, EventArgs e)
        {
            lbDescriptionM.Text = MinionHouse.Clean();
            pbM.Image = null;
        }

        private void btnChangeClothesM_Click(object sender, EventArgs e)
        {
            Color outfitColor = new Color();
            if (colorDialog1.ShowDialog() == DialogResult.OK)
            {
                outfitColor = colorDialog1.Color;
            }
            lbDescriptionM.Text = MinionCharacter.ChangeClothes(outfitColor);
        }

        private void btnTakeShowerM_Click(object sender, EventArgs e)
        {
            if (rbHotM.Checked == true)
            {
                lbDescriptionM.Text = MinionCharacter.TakeShower(rbHotM.Text);
            }
            else if (rbColdM.Checked == true)
            {
                lbDescriptionM.Text = MinionCharacter.TakeShower(rbColdM.Text);
            }
            else
            {
                MessageBox.Show("You have to select the water type first!");
            }
        }

        private void btnGoShoppingS_Click(object sender, EventArgs e)
        {
            try
            {
                lbDescriptionS.Text = SmurfFood.GoShopping(Convert.ToInt32(tbAmountS.Text));
            }
            catch (Exception)
            {
                MessageBox.Show("You have to write the amount of money!");
            }
}

        private void btnCookS_Click(object sender, EventArgs e)
        {
            lbDescriptionS.Text = SmurfFood.Cook();
        }

        private void btnBuildS_Click(object sender, EventArgs e)
        {
            pbS.Image = Image.FromFile("../Resources/smurfhouse.jpg");
            lbDescriptionS.Text = SmurfHouse.Build();
        }

        private void btnDecorateS_Click(object sender, EventArgs e)
        {
            lbDescriptionS.Text = SmurfHouse.Decorate();
        }

        private void btnCleanS_Click(object sender, EventArgs e)
        {
            lbDescriptionS.Text = SmurfHouse.Clean();
        }

        private void btnChangeClothesS_Click(object sender, EventArgs e)
        {
            Color outfitColor = new Color();
            if (colorDialog1.ShowDialog() == DialogResult.OK)
            {
                outfitColor = colorDialog1.Color;
            }
            lbDescriptionS.Text = SmurfCharacter.ChangeClothes(outfitColor);
        }

        private void btnTakeShowerS_Click(object sender, EventArgs e)
        {
