using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace AbstractFactoryCarDealer
{
    public abstract class SUV
    {
        protected string brand;

        public SUV(string brand)
        {
            this.brand = brand;
        }
        public override string ToString()
        {
            return brand + " " + Cars.SUV.ToString();
        }
    }
}
